renative.json API Reference

sdks

Define paths to your SDK Configurations

{
    "sdks": {
        "ANDROID_SDK": "",
        "ANDROID_NDK": "",
        "IOS_SDK": "",
        "TIZEN_SDK": "",
        "WEBOS_SDK": "",
        "KAIOS_SDK": ""
    }
}

workspaceID

Current workspace this project belongs to

{
    "workspaceID": "rnv"
}

paths

Define custom paths for RNV to look into

{
    "paths": {
        "entryDir": "./",
        "platformAssetsDir": "./platformAssets",
        "platformBuildsDir": "./platformBuilds"
    }
}

defaults

Default system config for this project

{
    "defaults": {
        "supportedPlatforms": [
            "ios",
            "android",
            "androidtv",
            "web",
            "macos",
            "tvos",
            "androidwear"
        ],
        "template": "",
        "schemes": {},
        "targets": {}
    }
}

pipes

To avoid rnv building buildHooks/src every time you can specify which specific pipes should trigger recompile of buildHooks

{
    "pipes": [
        "configure:after",
        "start:before",
        "deploy:after",
        "export:before",
        "export:after"
    ]
}
